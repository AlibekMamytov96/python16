ssh-keygen - это нужно для генерации ssh ключа
.ssh - это скрытая директория ssh настроек
id_rsa.pub - в ней хранится открытый ключ

git - система контроля версий
git init - initialization
.git - создается скрытая папка

git add <file_name> <.> - для проверки статуса
git commit -m "some comment" - он фикструет файл и создает хэш сумму

git remote add origin git@github.com:AlibekMamytov96/python16.git - связывает с удаленным сервером


git clone - просто клон проекта из удаленного сервера

git config --global user.name 'Alibek'
git config --global user.email alibek24galaxy@gmail.com
